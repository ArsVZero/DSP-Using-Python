Performing Statistical Operations on Signals

# Importing Experimental Signals
from mysignals import InputSignal_f32_1kHz_15kHz, Impulse_response, _640_points_ecg_


class StatisticalOperation:
    signal_mean = 0.0
    signal_variance = 0.0
    signal_standard_deviation = 0.0

    def mean_calculation(self, signal_source):
        for i in range(len(signal_source)):
            self.signal_mean += signal_source[i]
        self.signal_mean = self.signal_mean / len(signal_source)

        return self.signal_mean

    def variance_calculation(self, signal_source):
        for i in range(len(signal_source)):
            self.signal_mean += signal_source[i]
        self.signal_mean = self.signal_mean / len(signal_source)

        for i in range(len(signal_source)):
            self.signal_variance = self.signal_variance + (signal_source[i] - self.signal_mean) ** 2
        signal_variance = self.signal_variance / len(signal_source)

        return signal_variance

    def standard_deviation_calculation(self, signal_source):
        for i in range(len(signal_source)):
            self.signal_mean += signal_source[i]
        self.signal_mean = self.signal_mean / len(signal_source)

        for i in range(len(signal_source)):
            self.signal_variance = self.signal_variance + (signal_source[i] - self.signal_mean) ** 2
        signal_variance = self.signal_variance / len(signal_source)

        signal_standard_deviation = signal_variance ** 0.5

        return signal_standard_deviation

# Discrete Fourier Transformation Algorithm with Magnitude Analysis
def discrete_fourier_transform_magnitude(signal_source):
    global complex_part
    global real_part
    global magnitude

    complex_part = [None] * int((len(signal_source) / 2))
    real_part = [None] * int((len(signal_source) / 2))
    magnitude = [None] * int((len(signal_source) / 2))

    for i in range(int((len(signal_source) / 2))):
        real_part[i] = 0
        complex_part[i] = 0

    for j in range(int((len(signal_source) / 2))):
        for k in range(len(signal_source)):
            real_part[j] = real_part[j] + signal_source[k] * cos(2 * pi * j * k / len(signal_source))
            complex_part[j] = complex_part[j] - signal_source[k] * sin(2 * pi * j * k / len(signal_source))

    for x in range(int((len(signal_source) / 2))):
        magnitude[x] = sqrt(pow(real_part[x], 2) + pow(complex_part[x], 2))


# Inverse Discrete Fourier Transformation Algorithm
def inverse_discrete_fourier_transform(real, complex):
    global inverse_discrete_fourier_transformed_signal
    inverse_discrete_fourier_transformed_signal = [None] * (len(real) * 2)

    for i in range(len(real) * 2):
        inverse_discrete_fourier_transformed_signal[i] = 0

    for j in range(len(real)):
        real[j] /= len(real)
        complex[j] /= len(real)

    for k in range(len(real)):
        for x in range(len(real) * 2):
            inverse_discrete_fourier_transformed_signal[x] = inverse_discrete_fourier_transformed_signal[x] + real[
                k] * cos(
                2 * pi * k * x / (len(real) * 2))
            inverse_discrete_fourier_transformed_signal[x] = inverse_discrete_fourier_transformed_signal[x] + complex[
                k] * sin(
                2 * pi * k * x / (len(real) * 2))

Creating Custom Butterworth Band Pass Filter

def butter_band_pass_function(lower_cut_off, higher_cut_off, sampling_rate, order=5):
    nyquist_rate = 0.5 * sampling_rate
    lower_band = lower_cut_off / nyquist_rate
    higher_band = higher_cut_off / nyquist_rate
    butter_worth, analog = signal.butter(order, [lower_band, higher_band], btype='band')
    return butter_worth, analog


def butter_band_pass_filter(data, lower_cut_off, higher_cut_off, sampling_rate, order=5):
    butter_worth, analog = butter_band_pass_function(lower_cut_off, higher_cut_off, sampling_rate, order=order)
    filtered = signal.lfilter(butter_worth, analog, data)
    return filtered
